## 🛠️ 技術實現方案

### 必須新增的功能

#### 1. 用戶等級管理
```typescript
// lib/types.ts - 新增用戶類型
export type UserPlan = 'trial' | 'free' | 'pro' | 'enterprise'

export interface User {
  lineUserId: string
  displayName: string
  plan: UserPlan
  trialStartDate?: Date
  trialExpiredAt?: Date
  subscriptionExpiredAt?: Date
  maxCards: number
  allowedTemplates: string[]
}

// lib/auth/planChecker.ts - 權限檢查
export function checkUserPermissions(user: User) {
  const now = new Date()
  
  // 檢查試用期
  if (user.plan === 'trial' && user.trialExpiredAt && now > user.trialExpiredAt) {
    return {
      canEdit: false,
      canCreateNew: false,
      message: "試用期已結束，請升級繼續使用"
    }
  }
  
  return {
    canEdit: true,
    canCreateNew: user.maxCards > getUserCardCount(user.lineUserId),
    maxCards: user.maxCards
  }
}
```
